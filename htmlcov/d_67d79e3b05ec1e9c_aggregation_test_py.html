32from scipy.sparse.linalg import svds

34# python implementation for aggregation 

35def aggregate(data, sorting="pca", tol=0.5, early_stopping=False): # , verbose=1 

36 """Aggregate the data 

37 

38 Parameters 

39 ---------- 

40 data : numpy.ndarray 

41 The input that is array-like of shape (n_samples,). 

42 

43 sorting : str 

44 The sorting method for aggregation, default='pca', other options: 'norm-mean', 'norm-orthant'. 

45 

46 tol : float 

47 The tolerance to control the aggregation. if the distance between the starting point  

48 of a group and another data point is less than or equal to the tolerance, 

49 the point is allocated to that group.  

50 

51 Returns 

52 ------- 

53 labels (numpy.ndarray) :  

54 The group categories of the data after aggregation. 

55  

56 splist (list) :  

57 The list of the starting points. 

58  

59 nr_dist (int) : 

60 The number of pairwise distance calculations. 

61 """ 

62 

63 splist = list() # store the starting points 

64 len_ind = data.shape[0] 

65 

66 if sorting == "norm-mean" or sorting == "norm-orthant": 

67 c_data = data.copy() 

68 sort_vals = np.linalg.norm(c_data, ord=2, axis=1) 

69 ind = np.argsort(sort_vals) 

70 

71 elif sorting == "pca": 

72 c_data = data - data.mean(axis=0) 

73 if data.shape[1]>1: 

74 U1, s1, _ = svds(c_data, k=1, return_singular_vectors="u") 

75 sort_vals = U1[:,0]*s1[0] 

76 # print( U1, s1, _) 

77 else: 

78 sort_vals = c_data[:,0] 

79 sort_vals = sort_vals*np.sign(-sort_vals[0]) # flip to enforce deterministic output 

80 ind = np.argsort(sort_vals) 

81 

82 else: # no sorting 

83 sort_vals = np.zeros(len_ind) # useless, blank 

84 ind = np.arange(len_ind) 

85 

86 lab = 0 

87 labels = [-1]*len_ind 

88 nr_dist = 0 

89 

90 for i in range(len_ind): 

91 sp = ind[i] 

92 if labels[sp] >= 0: 

93 continue 

94 else: 

95 clustc = c_data[sp,:] 

96 labels[sp] = lab 

97 num_group = 1 

98 

99 for j in ind[i:]: 

100 if labels[j] >= 0: 

101 continue 

102 

103 if early_stopping: 

104 if (sort_vals[j] - sort_vals[sp] > tol): 

105 break 

106 

107 dat = clustc - c_data[j,:] 

108 dist = np.inner(dat, dat) 

109 nr_dist += 1 

110 

111 if dist <= tol**2: 

112 num_group += 1 

113 labels[j] = lab 

114 

115 splist.append([sp, lab] + [num_group] + list(data[sp,:]) ) 

116 lab += 1 

117 

118 # if verbose == 1: 

119 # print("aggregate {} groups".format(len(np.unique(labels)))) 

120 

121 return np.array(labels), splist, nr_dist
